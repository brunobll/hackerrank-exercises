NUMBER LINE JUMPERS 
-> Código que calcula se os cangurus conseguem atingir a mesma posição na mesma quantidade de pulos

#include<iostream>
using namespace std; 

int main() {
    unsigned int x1, v1; 
    unsigned int x2, v2;
    
    cin >> x1 >> v1 >> x2 >> v2;
    
    if(x1 > x2){
        if(v1 > v2){
            cout << "NO";
        }
        else if(v1 < v2){
            int qtdOfJumps1 = 0, qtdOfJumps2 = 0;
            bool samePosition = false;
            
            while(samePosition == false){
                x1 += v1;
                qtdOfJumps1++;
                x2 += v2;
                qtdOfJumps2++;
                
                cout << endl;
                cout << x1 << " " << x2;
                
                if(x1 == x2){
                    if(qtdOfJumps1 == qtdOfJumps2){
                        cout << "YES";
                        samePosition = true;
                        break;
                    }
                }
            }
        }

    }
    else if(x2 > x1){
        if(v2 > v1){
            cout << "NO";
        }
        
        else if(v2 < v1){
            int qtdOfJumps1 = 0, qtdOfJumps2 = 0;
            bool samePosition = false;
            
            while(samePosition == false){
                x1 += v1;
                qtdOfJumps1++;
                x2 += v2;
                qtdOfJumps2++;
                
                cout << endl;
                cout << x1 << " " << x2;
                
                if(x1 == x2){
                    if(qtdOfJumps1 == qtdOfJumps2){
                        cout << "YES";
                        samePosition = true;
                        break;
                    }
                }
            }
        }
    }

    
    return 0;
}
